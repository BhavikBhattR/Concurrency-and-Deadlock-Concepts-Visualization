struct semaphore
{
int value;
Queue type L;
}
DOWN SEMAPHORE

Down(semaphore S)
{
S.value=S.value-1;
if(S.value<0) {
put process(PCB) in Queue L
sleep();
}
else
return;
}
UP SEMAPHORE

Up(semaphore S)
{
S.value=S.value+1;
if(S.value<=0) {;
select a process from Queue L
wake up();
}
else
return;
}
